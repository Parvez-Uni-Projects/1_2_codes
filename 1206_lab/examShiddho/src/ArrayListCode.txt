// ArrayList Search

import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Scanner;

public class ArrayListTest {
    public static void main(String[] args) {
        ArrayList<Student> students = new ArrayList<>();
        try{
            Scanner sc = new Scanner(new File("files/data.txt"));

            while(sc.hasNext()){
                int id = sc.nextInt();
                String name = sc.next();
                double cgpa = sc.nextDouble();

                Student s = new Student(name, id, cgpa);
                students.add(s);
            }



            Scanner sc1 = new Scanner(System.in);
            String nameToFind = sc1.next();
            int found = 0;

            for(Student st: students){
                if(nameToFind.equals(st.name)){
                    System.out.println(st.name + " " + st.id + " " + st.cgpa);
                    found = 1;
                }
            }

            if(found == 0){
                System.out.println("Name not found!");
            }

        }
        catch (IOException e){
            e.printStackTrace();
        }
    }
}







// ArrayList Sorting using Comparable interface


class Student implements Comparable<Student> {
    String name;
    int id;
    double cgpa;

    Student(String name, int id, double cgpa){
        this.name = name;
        this.id = id;
        this.cgpa = cgpa;
    }

    @Override
    public int compareTo(Student o) {
        String n1 = this.name;
        String n2 = o.name;

        return n2.compareTo(n1);

        // returns negative: this object goes before o object
        // returns positive: o object goes before this object
        // returns 0: this object and o object are same.
    }
}




import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Scanner;

public class ArrayListTest {
    public static void main(String[] args) {
        ArrayList<Student> students = new ArrayList<>();
        try{
            Scanner sc = new Scanner(new File("files/data.txt"));

            while(sc.hasNext()){
                int id = sc.nextInt();
                String name = sc.next();
                double cgpa = sc.nextDouble();

                Student s = new Student(name, id, cgpa);
                students.add(s);
            }

            Collections.sort(students);

            for(Student st: students){
                System.out.println(st.name + " " + st.id + " " + st.cgpa);
            }
        }
        catch (IOException e){
            e.printStackTrace();
        }
    }
}
